105 void Reactor::Tick() {
106  // The following code must go in the Tick so they fire on the time step
107  // following the cycle_step update - allowing for the all reactor events to
108  // occur and be recorded on the "beginning" of a time step. Another reason
109  // they
110  // can't go at the beginnin of the Tock is so that resource exchange has a
111  // chance to occur after the discharge on this same time step.
112 
113  if (retired()) {
114  Record("RETIRED", "");
115 
116  if (context()->time() == exit_time()) { // only need to transmute once
117  Transmute(ceil(static_cast<double>(n_assem_core) / 2.0));
118  }
119  while (core.count() > 0) {
120  if (!Discharge()) {
121  break;
122  }
123  }
124  // in case a cycle lands exactly on our last time step, we will need to
125  // burn a batch from fresh inventory on this time step. When retired,
126  // this batch also needs to be discharged to spent fuel inventory.
127  while (fresh.count() > 0 && spent.space() >= assem_size) {
128  spent.Push(fresh.Pop());
129  }
130  return;
131  }
132 
133  if (cycle_step == cycle_time) {
134  Transmute();
135  Record("CYCLE_END", "");
136  }
137 
138  if (cycle_step >= cycle_time && !discharged) {
139  discharged = Discharge();
140  }
141  if (cycle_step >= cycle_time) {
142  Load();
143  }
144 
145  int t = context()->time();
146 
147  // update preferences
148  for (int i = 0; i < pref_change_times.size(); i++) {
149  int change_t = pref_change_times[i];
150  if (t != change_t) {
151  continue;
152  }
153 
154  std::string incommod = pref_change_commods[i];
155  for (int j = 0; j < fuel_incommods.size(); j++) {
156  if (fuel_incommods[j] == incommod) {
157  fuel_prefs[j] = pref_change_values[i];
158  break;
159  }
160  }
161  }
162 
163  // update recipes
164  for (int i = 0; i < recipe_change_times.size(); i++) {
165  int change_t = recipe_change_times[i];
166  if (t != change_t) {
167  continue;
168  }
169 
170  std::string incommod = recipe_change_commods[i];
171  for (int j = 0; j < fuel_incommods.size(); j++) {
172  if (fuel_incommods[j] == incommod) {
173  fuel_inrecipes[j] = recipe_change_in[i];
174  fuel_outrecipes[j] = recipe_change_out[i];
175  break;
176  }
177  }
178  }

184  std::set<RequestPortfolio<Material>::Ptr> ports;
185  Material::Ptr m;
186 
187  // second min expression reduces assembles to amount needed until
188  // retirement if it is near.
189  int n_assem_order = n_assem_core - core.count() + n_assem_fresh - fresh.count();
190 
191  if (exit_time() != -1) {
192  // the +1 accounts for the fact that the reactor is alive and gets to
193  // operate during its exit_time time step.
194  int t_left = exit_time() - context()->time() + 1;
195  int t_left_cycle = cycle_time + refuel_time - cycle_step;
196  double n_cycles_left = static_cast<double>(t_left - t_left_cycle) /
197  static_cast<double>(cycle_time + refuel_time);
198  n_cycles_left = ceil(n_cycles_left);
199  int n_need = std::max(0.0, n_cycles_left * n_assem_batch - n_assem_fresh);
200  n_assem_order = std::min(n_assem_order, n_need);
201  }
202 
203  if (n_assem_order == 0) {
204  return ports;
205  } else if (retired()) {
206  return ports;
207  }
208 
209  for (int i = 0; i < n_assem_order; i++) {
210  RequestPortfolio<Material>::Ptr port(new RequestPortfolio<Material>());
211  std::vector<Request<Material>*> mreqs;
212  for (int j = 0; j < fuel_incommods.size(); j++) {
213  std::string commod = fuel_incommods[j];
214  double pref = fuel_prefs[j];
215  Composition::Ptr recipe = context()->GetRecipe(fuel_inrecipes[j]);
216  m = Material::CreateUntracked(assem_size, recipe);
217  Request<Material>* r = port->AddRequest(m, this, commod, pref, true);
