122 Material::Ptr SepMaterial(std::map<int, double> effs, Material::Ptr mat) {
123  CompMap cm = mat->comp()->mass();
124  cyclus::compmath::Normalize(&cm, mat->quantity());
125  double tot_qty = 0;
126  CompMap sepcomp;
127 
128  CompMap::iterator it;
129  for (it = cm.begin(); it != cm.end(); ++it) {
130  int nuc = it->first;
131  int elem = (nuc / 10000000) * 10000000;
132  double eff = 0;
133  if (effs.count(nuc) > 0) {
134  eff = effs[nuc];
135  } else if (effs.count(elem) > 0) {
136  eff = effs[elem];
137  } else {
138  continue;
139  }
140 
141  double qty = it->second;
142  double sepqty = qty * eff;
143  sepcomp[nuc] = sepqty;
144  tot_qty += sepqty;
145  }
146 
147  Composition::Ptr c = Composition::CreateFromMass(sepcomp);
148  return Material::CreateUntracked(tot_qty, c);
