995 void Reactor::Tick() {
996  // The following code must go in the Tick so they fire on the time step
997  // following the cycle_step update - allowing for the all reactor events to
998  // occur and be recorded on the "beginning" of a time step. Another reason
999  // they
1000  // can't go at the beginnin of the Tock is so that resource exchange has a
1001  // chance to occur after the discharge on this same time step.
1002 
1003  if (retired()) {
1004  Record("RETIRED", "");
1005 
1006  if (context()->time() == exit_time()) { // only need to transmute once
1007  Transmute(ceil(static_cast<double>(n_assem_core) / 2.0));
1008  }
1009  while (core.count() > 0) {
1010  if (!Discharge()) {
1011  break;
1012  }
1013  }

1015  // burn a batch from fresh inventory on this time step. When retired,
